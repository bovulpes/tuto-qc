560 class Varinfo : public IDisplay {
561  Location loc_;
562  ident id_;
563  int offset_;
564 
565  public:
573  Varinfo(Location loc, ident id, int offset = -1)
574  : loc_(loc), id_(id), offset_(offset) {}
575 
583  std::vector<idx> as_creg(Context& ctx) const {
584  auto reg = dynamic_cast<const Register*>(ctx.lookup(id_, loc_));
585 
586  if (reg == nullptr || reg->quantum_) {
587  std::cerr << loc_ << ": Identifier " << id_
588  << " does not refer to a bit register\n";
589  throw exception::SemanticError("qpp::qasm::Varinfo::as_creg()");
590  }
591 
592  if (offset_ == -1) {
593  // creg
594  return std::vector<idx>{reg->indices_};
595  } else {
596  // creg deref
597 
598  // check register size
599  if ((idx) offset_ >= reg->indices_.size()) {
600  std::cerr << loc_ << ": Index out of bounds\n";
601  throw exception::SemanticError("qpp::qasm::Varinfo::as_qreg()");
602  }
603 
604  return std::vector<idx>{reg->indices_[offset_]};
605  }
606  }
607 
616  std::vector<idx> as_qreg(Context& ctx) const {
617  if (offset_ == -1) {
618  // qubit or qreg
619  auto tmp = ctx.lookup(id_, loc_);
620 
621  if (auto qubit = dynamic_cast<const Qubit*>(tmp)) {
622  return std::vector<idx>{qubit->index_};
623  } else {
624  auto reg = dynamic_cast<const Register*>(tmp);
625  // check register type
626  if (reg == nullptr || !reg->quantum_) {
627  std::cerr << loc_ << ": Identifier " << id_
628  << " does not refer to a qubit register\n";
629  throw exception::SemanticError(
630  "qpp::qasm::Varinfo::as_qreg()");
631  }
632 
633  return std::vector<idx>{reg->indices_};
634  }
635  } else {
636  // qreg deref
637  auto reg = dynamic_cast<const Register*>(ctx.lookup(id_, loc_));
638 
639  // check register type
640  if (reg == nullptr || !reg->quantum_) {
641  std::cerr << loc_ << ": Identifier " << id_
642  << " does not refer to a qubit register\n";
643  throw exception::SemanticError("qpp::qasm::Varinfo::as_qreg()");
644  }
645 
646  // check register size
647  if ((idx) offset_ >= reg->indices_.size()) {
648  std::cerr << loc_ << ": Index out of bounds\n";
649  throw exception::SemanticError("qpp::qasm::Varinfo::as_qreg()");
650  }
651 
652  return std::vector<idx>{reg->indices_[offset_]};
653  }
654  }
655 
659  std::ostream& display(std::ostream& os) const {
660  os << id_;
661  if (offset_ != -1)
662  os << "[" << offset_ << "]";
663 
664  return os;
665  }

938  }

954 class UGate final : public Gate {
955  ExprPtr theta_;
956  ExprPtr phi_;
957  ExprPtr lambda_;
958 
959  Varinfo arg_;
960  public:
970  UGate(Location loc, ExprPtr theta, ExprPtr phi, ExprPtr lambda, Varinfo arg)
971  : Gate(loc), theta_(std::move(theta)), phi_(std::move(phi)),
972  lambda_(std::move(lambda)), arg_(arg) {}
973 
977  void evaluate(Context& ctx) const {
978  auto theta = theta_->evaluate(ctx);
979  auto phi = phi_->evaluate(ctx);
980  auto lambda = lambda_->evaluate(ctx);
981  auto args = arg_.as_qreg(ctx);
982  auto circuit = ctx.get_circuit();
983 
984  // generate the matrix
985  cmat u{cmat::Zero(2, 2)};
986  u << cos(theta / 2) * std::exp(-1_i * (phi + lambda) / 2.0),
987  -(sin(theta / 2)) * std::exp(-1_i * (phi - lambda) / 2.0),
988  sin(theta / 2) * std::exp(1_i * (phi - lambda) / 2.0),
989  cos(theta / 2) * std::exp(1_i * (phi + lambda) / 2.0);
990 
991  // apply the gate
992  for (auto i : args) {
993  if (ctx.ccontrolled()) {
994  circuit->cCTRL(u, ctx.get_cctrls(), i, ctx.get_shift(),
995  "Controlled-U");
996  } else {
997  circuit->gate(u, i, "U");
998  }
999  }
1000  }
1001 
1005  std::ostream& display(std::ostream& os) const {
1006  os << "U(" << *theta_ << "," << *phi_ << "," << *lambda_ << ") " << arg_
1007  << ";\n";
1008  return os;
1009  }

1017 class CNOTGate final : public Gate {
1018  Varinfo ctrl_;
1019  Varinfo tgt_;
1020 
1021  public:
1029  CNOTGate(Location loc, Varinfo ctrl, Varinfo tgt)
1030  : Gate(loc), ctrl_(ctrl), tgt_(tgt) {}
1031 
1035  void evaluate(Context& ctx) const {
1036  auto ctrls = ctrl_.as_qreg(ctx);
1037  auto tgts = tgt_.as_qreg(ctx);
1038  auto circuit = ctx.get_circuit();
1039 
1040  // different combinations of controls and targets
1041  if (ctrls.size() == 1 && tgts.size() == 1) {
1042  if (ctx.ccontrolled()) {
1043  std::vector<idx> tmp{ctrls[0], tgts[0]};
1044  circuit->cCTRL_custom(gt.CNOT, ctx.get_cctrls(), tmp,
1045  ctx.get_shift(), "CX");
1046  } else {
1047  circuit->gate(gt.CNOT, ctrls[0], tgts[0], "CX");
1048  }
1049  } else if (ctrls.size() > 1 && tgts.size() == 1) {
1050  for (idx i = 0; i < ctrls.size(); i++) {
1051  if (ctx.ccontrolled()) {
1052  std::vector<idx> tmp{ctrls[i], tgts[0]};
1053  circuit->cCTRL_custom(gt.CNOT, ctx.get_cctrls(), tmp,
1054  ctx.get_shift(), "CX");
1055  } else {
1056  circuit->gate(gt.CNOT, ctrls[i], tgts[0], "CX");
1057  }
1058  }
1059  } else if (ctrls.size() == 1 && tgts.size() > 1) {
1060  for (idx i = 0; i < tgts.size(); i++) {
1061  if (ctx.ccontrolled()) {
1062  std::vector<idx> tmp{ctrls[0], tgts[i]};
1063  circuit->cCTRL_custom(gt.CNOT, ctx.get_cctrls(), tmp,
1064  ctx.get_shift(), "CX");
1065  } else {
1066  circuit->gate(gt.CNOT, ctrls[0], tgts[i], "CX");
1067  }
1068  }
1069  } else if (ctrls.size() == tgts.size()) {
1070  for (idx i = 0; i < ctrls.size(); i++) {

1174  auto gate = dynamic_cast<const Circuit*>(ctx.lookup(id_, loc_));
1175 
1176  // check gate type
1177  if (gate == nullptr) {
1178  std::cerr << loc_ << ": Identifier " << id_
1179  << " does not refer to a gate declaration\n";
1180  throw exception::SemanticError(
1181  "qpp::qasm::DeclaredGate::evaluate()");
1182  }
1183 
1184  // check argument lengths
1185  if (c_args_.size() != gate->c_params_.size()) {
1186  std::cerr << loc_ << ": " << id_ << " expects "
1187  << gate->c_params_.size();
1188  std::cerr << " classic arguments, got " << c_args_.size() << "\n";
1189  throw exception::SemanticError(
1190  "qpp::qasm::DeclaredGate::evaluate()");
1191  } else if (q_args_.size() != gate->q_params_.size()) {
1192  std::cerr << loc_ << ": " << id_ << " expects "
1193  << gate->q_params_.size();
1194  std::cerr << " quantum arguments, got " << q_args_.size() << "\n";
1195  throw exception::SemanticError(
1196  "qpp::qasm::DeclaredGate::evaluate()");
1197  }
1198 
1199  // evaluate arguments
1200  std::vector<double> c_args(c_args_.size());
1201  std::vector<std::vector<idx>> q_args(q_args_.size());
1202  for (idx i = 0; i < c_args_.size(); i++)
1203  c_args[i] = c_args_[i]->evaluate(ctx);
1204  for (idx i = 0; i < q_args_.size(); i++)
1205  q_args[i] = q_args_[i].as_qreg(ctx);
1206 
1207  // map gate across registers
1208  idx mapping_size = 1;
1209  std::vector<bool> mapped(q_args.size(), false);
1210  for (idx i = 0; i < q_args.size(); i++) {
1211  if (q_args[i].size() > 1) {
1212  mapped[i] = true;
1213  if (mapping_size == 1) {
1214  mapping_size = q_args[i].size();
1215  } else if (mapping_size != q_args[i].size()) {
1216  std::cerr << loc_ << ": Registers have different lengths\n";
1217  throw exception::SemanticError(
1218  "qpp::qasm::DeclaredGate::evaluate()");
1219  }
1220  }
1221  }
1222 
1223  auto it = known_matrices.find(id_);
1224  if (it != known_matrices.end()) {
1225  // apply the known matrix directly
1226  auto circuit = ctx.get_circuit();
1227  auto mat = it->second(c_args);
1228 
1229  // map the gate accross registers
1230  for (idx j = 0; j < mapping_size; j++) {
1231  // map virtual qubits to physical qubits
1232  std::vector<idx> mapped_args(q_args.size());
1233  for (idx i = 0; i < q_args.size(); i++) {
1234  mapped_args[i] = mapped[i] ? q_args[i][j] : q_args[i][0];
1235  }
1236 
1237  // apply (possibly classical controlled) gate
1238  if (ctx.ccontrolled()) {
1239  circuit->cCTRL_custom(mat, ctx.get_cctrls(), mapped_args,
1240  ctx.get_shift(), id_);
1241  } else {
1242  circuit->gate_custom(mat, mapped_args, id_);
1243  }
1244  }
1245  } else {
1246  // push classical arguments onto a new scope
1247  ctx.enter_scope();
1248  for (idx i = 0; i < c_args.size(); i++) {
1249  ctx.set(gate->c_params_[i],
1250  std::unique_ptr<Value>(new Number(c_args[i])));
1251  }
1252 
1253  // map the gate
1254  for (idx j = 0; j < mapping_size; j++) {
1255  ctx.enter_scope();
1256  for (idx i = 0; i < q_args.size(); i++) {
1257  ctx.set(gate->q_params_[i],
1258  std::unique_ptr<Value>(new Qubit(
1259  mapped[i] ? q_args[i][j] : q_args[i][0])));
1260  }
1261 
1262  // evaluate the gate
1263  for (auto it = gate->body_.begin(); it != gate->body_.end();
1264  it++) {
1265  (*it)->evaluate(ctx);
1266  }
1267 
1268  ctx.exit_scope();
1269  }
1270  ctx.exit_scope();
1271  }
1272  }
1273 
1277  std::ostream& display(std::ostream& os) const {
1278  os << id_;
1279  if (c_args_.size() > 0) {
1280  os << "(";
1281  for (auto it = c_args_.begin(); it != c_args_.end(); it++) {
1282  os << (it == c_args_.begin() ? "" : ",") << **it;
1283  }
1284  os << ")";
1285  }
1286  os << " ";
1287  for (auto it = q_args_.begin(); it != q_args_.end(); it++) {
1288  os << (it == q_args_.begin() ? "" : ",") << *it;
1289  }
1290  os << ";\n";
1291  return os;
1292  }

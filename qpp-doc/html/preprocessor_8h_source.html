100  bool add_target_file(const std::string& file_path) {

142  Token next_token() {
143  if (current_lexer_ == nullptr) {
144  return Token(Location(), Token::Kind::eof, "");
145  }
146  auto token = current_lexer_->next_token();
147  if (token.is(Token::Kind::kw_include)) {
148  handle_include();
149  token = current_lexer_->next_token();
150  } else if (token.is(Token::Kind::eof)) {
151  if (!lexer_stack_.empty()) {
152  current_lexer_ = std::move(lexer_stack_.back());
153  lexer_stack_.pop_back();
154  token = current_lexer_->next_token();
155  } else {
156  current_lexer_ = nullptr;
157  }
158  }
159  return token;
160  }
161 
165  void print_all_tokens() {
166  Token current = next_token();
167 
168  while (!current.is(Token::Kind::eof)) {
169  current.location().display(std::cout);
170  std::cout << ": " << current << " " << (std::string) current
171  << "\n";
172 
173  current = next_token();
174  }
175  }
176 
177  private:
185  void handle_include() {
186  auto token = current_lexer_->next_token();
187  if (token.is_not(Token::Kind::string)) {
188  std::cerr << "Error: Include must be followed by a file name\n";
189  return;
190  }
191 
192  std::string target = token;
193  token = current_lexer_->next_token();
194  if (token.is_not(Token::Kind::semicolon)) {
195  std::cerr << "Warning: Missing a ';'\n";
196  }
197  if (add_target_file(target)) {
198  return;
199  } else if (target == "qelib1.inc") {
200  add_target_stream(std::shared_ptr<std::istream>(
201  new std::stringstream(std_include)),
202  "qelib1.inc");
203  return;
204  } else {
205  std::cerr << "Error: Couldn't open file " << target << "\n";
206  }
207  }
208 };
